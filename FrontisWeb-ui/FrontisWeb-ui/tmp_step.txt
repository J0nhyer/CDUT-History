export function getPersonData(personId) {
  const person = allPersonsData[personId];
  
  if (!person) {
    console.error(`æœªæ‰¾åˆ°äººç‰©: ${personId}`);
    return null;
  }
  
  // åœ¨å¼€å‘ç¯å¢ƒä¸­è­¦å‘ŠæœªéªŒè¯çš„æ•°æ®
  if (process.env.NODE_ENV === 'development') {
    if (person.dataStatus !== 'completed') {
      console.warn(`âš ï¸ è­¦å‘Š: äººç‰© "${person.name}" çš„æ•°æ®çŠ¶æ€ä¸º "${person.dataStatus}"ï¼Œéœ€è¦éªŒè¯ï¼`);
    }
    
    // æ£€æŸ¥æ˜¯å¦æœ‰å¾…éªŒè¯çš„é¡¹ç›®
    const needsVerification = checkNeedsVerification(person);
    if (needsVerification.length > 0) {
      console.warn(`âš ï¸ "${person.name}" æœ‰ ${needsVerification.length} é¡¹å†…å®¹éœ€è¦éªŒè¯:`, needsVerification);
    }

  return person;
}

/**
 * æ£€æŸ¥éœ€è¦éªŒè¯çš„å†…å®¹
 */
function checkNeedsVerification(person) {
  const items = [];
  
  // æ£€æŸ¥æ—¶é—´è½´
  person.timeline?.forEach((event, index) => {
    if (event.needsVerification) {
      items.push(`æ—¶é—´è½´[${index}]: ${event.title}`);
    }
  });
  
  // æ£€æŸ¥ç”Ÿå¹³
  person.biography?.forEach((chapter, index) => {
    if (chapter.needsVerification) {
      items.push(`ç”Ÿå¹³[${index}]: ${chapter.title}`);
    }
  });
  
  // æ£€æŸ¥æˆå°±
  person.achievements?.forEach((achievement, index) => {
    if (achievement.needsVerification) {
      items.push(`æˆå°±[${index}]: ${achievement.title}`);
    }
  });
  
  // æ£€æŸ¥å…³ç³»
  person.relationships?.forEach((rel, index) => {
    if (rel.needsVerification) {
      items.push(`å…³ç³»[${index}]: ${rel.name}`);
    }
  });
  
  return items;
}
